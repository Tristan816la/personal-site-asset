## Install

```bash
npm install --save personalsiteassets
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'personalsiteassets'
import 'personalsiteassets/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
